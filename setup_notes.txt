#instructions for setting up folders and conda environment

##create a new qchem conda enviornment and folders
git clone https://github.com/icanswim/cosmosis.git
git clone https://github.com/icanswim/qchem.git
cd qchem
mkdir data
mkdir logs
mkdir models
conda env create -f environment.yml
conda activate qchem
jupyter lab

##manual creation of conda enviornment and folders
git clone https://github.com/icanswim/cosmosis.git
git clone https://github.com/icanswim/qchem.git
cd qchem
mkdir data
mkdir logs
mkdir models
conda create --name qchem -c pytorch -c nvidia python=3.10 pytorch torchvision torchaudio pytorch-cuda=11.7
conda activate qchem
conda install -c conda-forge matplotlib
conda install -c conda-forge scipy pandas scikit-learn
conda install -c conda-forge jupyterlab h5py git
conda instsll -c conda-forge numpy=1.24
conda install pyg -c pyg
jupyter lab
